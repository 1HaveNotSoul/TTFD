from collections import defaultdict
import time

user_command_times = defaultdict(list)

def rate_limit(user_id, max_commands=5, time_window=60):
    """–ú–∞–∫—Å–∏–º—É–º 5 –∫–æ–º–∞–Ω–¥ –≤ –º–∏–Ω—É—Ç—É"""
    now = time.time()

    # –£–¥–∞–ª—è–µ–º —Å—Ç–∞—Ä—ã–µ
    user_times = [t for t in user_times if now - t < time_window]
    
    if len(user_times) >= max_commands:
        return False
    
    user_times.append(now)
    user_command_times[user_id] = user_times
    return True
